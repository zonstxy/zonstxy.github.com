# 添加步骤📕

# 🎓STEP 1

拉取最新框架文件放置本地框架目录：\gameFrameworks\src\game\room\mj\view\bottom_mj_player_layer.luac

# 🎓STEP 2

重写major_player_layer.lua的adjustHandCardPos方法：

function majorPlayerLayer:adjustHandCardPos(offsetx, param)
    param=param or {}
    if not GameLogicCtrl.tagGameRuleConfig.isQiePai or param.needAnim==nil then
        --isQiePai字段用作兼容可选项，不需要可以去掉
        self:adjustHandCardPosNormal(offsetx, param)
    else
        self:adjustHandCardPosAnim(offsetx, param)
    end
end

在major_player_layer.lua下加入isDispatching字段，区分正常出牌和吃碰杠操作：

function majorPlayerLayer:onCreate(param)
    self.isDispatching=false       
end
function majorPlayerLayer:playNormalCardEx(pcard, param)
	self.isDispatching=true
end
function majorPlayerLayer:playRuleCard(data, ruletp, param)
	for k,v in pairs(delHandCardsIndex) do
        self.isDispatching=false
		self:removeHandCard(v)
	end
end

# 🎓STEP 3

写入major_player_layer.lua的adjustHandCardPosNormal方法：

function majorPlayerLayer:adjustHandCardPosNormal(offsetx, param)
     local myHhandCard = DeepCopy(self:getHandsCardsRef())
    local x = offsetx or 0
	x=x+self:getSidePositionByRuleCards()
	param = param or {}

	if IsMahjong3D() then
		x = x + 25
	end
	local sortCardFunc=self:getSortCardFunc()
	if sortCardFunc then
		local handcardValue = {}
		for i=1, #myHhandCard do 
			handcardValue[#handcardValue+1] = myHhandCard[i]:getValue() 
		end
		handcardValue = sortCardFunc(handcardValue)	
		local newMyHandCards = {}
		for i=1, #handcardValue do
			for j=1, #myHhandCard do
				if handcardValue[i] == myHhandCard[j]:getValue() then
					newMyHandCards[#newMyHandCards+1] = myHhandCard[j]
					table.remove(myHhandCard, j)
					break
				end
			end
		end
		myHhandCard = newMyHandCards
	end
	
	for k,v in pairs(myHhandCard) do
		v:setPos(x, param.touchUp and v:getPos().y or 0)
		v.upPos = cc.p(x, 30)
		v.normalPos = cc.p(x, 0)
		x=x+v:getSize().width+getMjCardResConfig("bottom").normalcard_offsetx
	end
	if not param.notDispatch and #myHhandCard%3 == 2 then
		local x = myHhandCard[#myHhandCard]:getPositionX()+self:getDispatchCardInterval()
		myHhandCard[#myHhandCard]:setPositionX(x)
		myHhandCard[#myHhandCard].upPos.x = x
		myHhandCard[#myHhandCard].normalPos.x = x
	end
end

写入major_player_layer.lua的adjustHandCardPosAnim方法：

function majorPlayerLayer:adjustHandCardPosAnim(offsetx, param)
    local myHhandCard = DeepCopy(self:getHandsCardsRef())
	local x = offsetx or 0
    local moveToIndex=0
    local movetoX=0
    local objCard={}
    
	param = param or {}
    local needAnim=param.needAnim and self.isDispatching
	x=x+self:getSidePositionByRuleCards()
	if IsMahjong3D() then
		x = x + 25
	end
    if needAnim then
        objCard=myHhandCard[#myHhandCard]
    end
	local sortCardFunc=self:getSortCardFunc()
	if sortCardFunc then
		local handcardValue = {}
		for i=1, #myHhandCard do 
			handcardValue[#handcardValue+1] = myHhandCard[i]:getValue() 
            myHhandCard[i].preIndex=i
		end
		handcardValue = sortCardFunc(handcardValue)	
		local newMyHandCards = {}
		for i=1, #handcardValue do
			for j=1, #myHhandCard do
                if needAnim then
                    if objCard:getValue()==handcardValue[i] then
                       moveToIndex=i 
                    end
                end
				if handcardValue[i] == myHhandCard[j]:getValue() then
					newMyHandCards[#newMyHandCards+1] = myHhandCard[j]
					table.remove(myHhandCard, j)
					break
				end
			end
		end
		myHhandCard = newMyHandCards
	end
    self.movePos={}
    for k,v in pairs(myHhandCard) do
        if (objCard==v) and needAnim and (#myHhandCard-moveToIndex)>=1 then
            movetoX=x
        else
			--[[if v.o_setPos~=nil then
				v.setPos=v.o_setPos
			end--]]
            self.movePos[k]={}
            self.movePos[k] = cc.p(x, param.touchUp and v:getPos().y or 0)
            logc("v的原始位置：",v.preIndex)
            if (#myHhandCard==k and not self.isDispatching) or not self.isDispatching then
                v:stopAllActions()
                v:setPosition(x,param.touchUp and v:getPos().y or 0)
                v.upPos = cc.p(x, 30)
                v.normalPos = cc.p(x, 0) 
            else
                v:stopAllActions()
                v:setPosition(x,param.touchUp and v:getPos().y or 0)
                v.upPos = cc.p(x, 30)
                v.normalPos = cc.p(x, 0) 
            end
            v:setRotation(0)
            --v.touchUp = nil
        end
        x=x+v:getSize().width+getMjCardResConfig("bottom").normalcard_offsetx
    end
    if  #myHhandCard%3 == 2 then
        local lex = myHhandCard[#myHhandCard].normalPos.x
         logc("move!!!!!!!!!!!!!!!!!!!!",lex)
        lex=lex+self:getDispatchCardInterval()
        logc("move!!!!!!!!!!!!!!!!!!!!",lex)
        myHhandCard[#myHhandCard]:setPositionX(lex)
        myHhandCard[#myHhandCard].upPos.x = lex
        myHhandCard[#myHhandCard].normalPos.x = lex
    end
    self:setHandsCardsRef(myHhandCard)
    --需要执行动画，并且牌的移动距离大于1张牌
    if needAnim and (#myHhandCard-moveToIndex)>=1 then
        local tempX,tempY=objCard:getPosition()
        local radian = 90*3.14159/180.0;
		local height  = objCard:getSize().height*0.85
		local q1x = tempX+(movetoX-tempX)/2.0
		--local q1 = cc.p(q1x, height + tempY+math.cos(radian)*q1x);		

		local q2x = tempX+(movetoX-tempX)*4/5
		--local q2 = cc.p(q2x, height + tempY+math.cos(radian)*q2x);
		local q1 = cc.p(q1x,height)
        local q2 = cc.p(movetoX,height)
		local cfg = {}
		cfg[#cfg+1] = q1
		cfg[#cfg+1] = q2
        cfg[#cfg+1] = cc.p(movetoX,0)
        local moveTime = 0.18
        local bezierto=cc.BezierTo:create(moveTime,cfg)
        local moveTo = cc.MoveBy:create(moveTime,cc.p((movetoX-tempX),0))
        local rotateAnim=cc.RotateBy:create(0.07,30)
        local rotateAnim2=cc.RotateBy:create(0.1,-30)
        local subAnim = cc.Spawn:create(cc.MoveBy:create(0.07,cc.p(0,height)),rotateAnim)
        local subAnim2 = cc.Spawn:create(cc.MoveBy:create(0.1,cc.p(0,-height)),rotateAnim2)
        local anim=cc.Sequence:create(subAnim,moveTo,subAnim2,cc.CallFunc:create(function ()
            objCard:setRotation(0)
            objCard.upPos = cc.p(movetoX, 30)
            objCard.normalPos = cc.p(movetoX, 0)
            objCard.setPos = objCard.o_setPos
			objCard.o_setPos=nil
        end),nil)
        
        --这里设置正在执行Action的标识，牌的touchBegan里根据他不设置Pos
        objCard:runAction(anim)
        objCard.o_setPos = objCard.setPos
        objCard.setPos = function (x, y)
        end
    end
end
